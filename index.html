/* 중심 (중력 포인트) */ let cx = innerWidth/2; let cy = innerHeight/2; /* 별 */ let stars=[]; for(let i=0;i<160;i++){ stars.push({ x:Math.random()*innerWidth, y:Math.random()*innerHeight, vx:(Math.random()-0.5)*0.3, vy:(Math.random()-0.5)*0.3, z:Math.random()*2+0.5, tw:Math.random()*100 }); } function drawBG(){ bgCtx.fillStyle="#02030a"; bgCtx.fillRect(0,0,bg.width,bg.height); stars.forEach(s=>{ /* 🔥 중력 */ let dx = cx - s.x; let dy = cy - s.y; let dist = Math.sqrt(dx*dx + dy*dy) + 0.1; let force = 0.05 / dist; s.vx += dx * force; s.vy += dy * force; /* 이동 */ s.x += s.vx * s.z; s.y += s.vy * s.z; /* 화면 밖 처리 */ if(s.x<0||s.x>bg.width||s.y<0||s.y>bg.height){ s.x=Math.random()*bg.width; s.y=Math.random()*bg.height; s.vx=0; s.vy=0; } /* 반짝 */ s.tw += 0.05; let alpha = 0.4 + Math.sin(s.tw)*0.2; /* 패럴랙스 (약하게!) */ let px = s.x + (mx-bg.width/2)*0.01*s.z; let py = s.y + (my-bg.height/2)*0.01*s.z; bgCtx.fillStyle=`rgba(255,255,255,${alpha})`; bgCtx.fillRect(px,py,s.z*1.5,s.z*1.5); /* 💥 낮은 확률 폭발 */ if(Math.random()<0.0004){ spawnExplosion(px,py); } }); /* 💥 파티클 */ particles.forEach(p=>{ p.x+=p.vx; p.y+=p.vy; p.life-=0.015; bgCtx.fillStyle=`rgba(255,180,80,${p.life*0.6})`; bgCtx.fillRect(p.x,p.y,2,2); }); particles = particles.filter(p=>p.life>0); }
